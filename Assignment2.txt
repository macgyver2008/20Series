public class Animal{
    /**
    * Color of animal.
    */
    private static String color = "";
    /**
    * Name of animal.
    */
    private static String name = "";
    /**
    * limited speed of animal.
    */
    private static int maxSpeed = 0;
    /**
    * Speed of animal.
    */
    private static int speed = 0;

    /**
     * @param fColor
     * @param fLicensePlate
     * @param fName
     * @param fMaxSpeed
     */
    public Animal(final String fColor, final String fName,
    final int fMaxSpeed) {
        this.color = fColor;
        this.name = fName;
        this.maxSpeed = fMaxSpeed;
        this.speed = 0;
    }

    /**
     * @param power
     * @param time
     * @return speed
     * which is animal's speed.
     */
    public int accelerate(final int power, final int time) {
        /*
        * Accelerate, speed = power * time
        * speed <= maxSpeed
        */
        speed += power * time;
        if (speed <= maxSpeed) {
            return speed;
        } else {
            speed = maxSpeed;
            return speed;
        }
    }

    /**
     * @param power
     * @param time
     * @return speed
     * which is car's speed.
     */
    public int brake(final int power, final int time) {
        // break, speed = speed - power * time
        // speed >= 0
       speed -= power * time;
        if (speed >= 0) {
            return speed;
        } else {
            speed = 0;
            return speed;
        }
    }


}
