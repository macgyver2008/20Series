public class Animal{
    /**
    * Color of animal.
    */
    private static String color = "";
    /**
    * Name of animal.
    */
    private static String name = "";
    /**
    * limited speed of animal.
    */
    private static int maxSpeed = 0;
    /**
    * Speed of animal.
    */
    private static int speed = 0;

    /**
     * @param fColor
     * @param fLicensePlate
     * @param fName
     * @param fMaxSpeed
     */
    public Animal(final String fColor, final String fName,
    final int fMaxSpeed) {
        this.color = fColor;
        this.name = fName;
        this.maxSpeed = fMaxSpeed;
        this.speed = 0;
    }

    /**
     * @param power
     * @param time
     * @return speed
     * which is animal's speed.
     */
    public int accelerate(final int power, final int time) {
        /*
        * Accelerate, speed = power * time
        * speed <= maxSpeed
        */
        speed += power * time;
        if (speed <= maxSpeed) {
            return speed;
        } else {
            speed = maxSpeed;
            return speed;
        }
    }

    /**
     * @param power
     * @param time
     * @return speed
     * which is car's speed.
     */
    public int brake(final int power, final int time) {
        // break, speed = speed - power * time
        // speed >= 0
       speed -= power * time;
        if (speed >= 0) {
            return speed;
        } else {
            speed = 0;
            return speed;
        }
    }


}

public final class UselessAnimalRacing {

    /**
     * Prevent instantiation.
     */
    private UselessAnimalRacing() {
        throw new IllegalStateException("Cannot be instantiated");
    }

    /**
     * The starting main() function.
     *
     * @param args No args will be used
     */
    public static void main(final String[] args) {

        // Create animals
        Hare hare = new Hare();
        Tortoise tortoise = new Tortoise();
        Random random = new Random();

        final int track = 1000;   // finish distance
        int hareDistance = track;
        int tortoiseDistance = track;

        int hareSleep = 0;        // hare sleeping turns
        int turtleSleep = 0;      // tortoise resting turns

        int turn = 1;

        System.out.println("=============================================");
        System.out.println("      Welcome to Silver Forest Racing!       ");
        System.out.println("=============================================");
        delay(2);

        hare.introduce();
        delay(2);
        tortoise.introduce();
        delay(2);

        System.out.println("\nRace Start!\n");

        // GAME LOOP
        while (hareDistance > 0 && tortoiseDistance > 0) {

            System.out.println("---------- TURN " + turn + " ----------");

            // 1) HARE TURN -------------------------------------------------
            if (hareSleep > 0) {
                System.out.println("Hare is sleeping... remaining turns: " + hareSleep);
                hareSleep--;
            } else {
                int power = random.nextInt(30) + 1; // 1~30
                int time = random.nextInt(3) + 1;   // 1~3

                hare.accelerate(power, time);
                hareDistance -= hare.getSpeed();

                // special skill chance
                int chance = random.nextInt(100) + 1;
                hareSleep = hare.specialSkill(chance);

                System.out.println("ðŸ° Hare speed: " + hare.getSpeed());
                System.out.println("ðŸ° Hare remaining distance: " + Math.max(hareDistance, 0));
            }

            // 2) TORTOISE TURN ----------------------------------------------
            if (turtleSleep > 0) {
                System.out.println("Tortoise is resting... remaining turns: " + turtleSleep);
                turtleSleep--;
            } else {
                int power = random.nextInt(30) + 1; // 1~30
                int time = random.nextInt(3) + 1;   // 1~3

                tortoise.accelerate(power, time);
                tortoiseDistance -= tortoise.getSpeed();

                // special skill chance
                int chance = random.nextInt(100) + 1;
                turtleSleep = tortoise.specialSkill(chance);

                System.out.println("Tortoise speed: " + tortoise.getSpeed());
                System.out.println("Tortoise remaining distance: " + Math.max(tortoiseDistance, 0));
            }

            // WIN CONDITION ---------------------------------------------------
            if (hareDistance <= 0 && tortoiseDistance <= 0) {
                System.out.println("\nIt's a TIE!!!");
                break;
            } else if (hareDistance <= 0) {
                System.out.println("\nHare wins the race!");
                break;
            } else if (tortoiseDistance <= 0) {
                System.out.println("\nTortoise wins the race!");
                break;
            }

            turn++;
            delay(1);
            System.out.println();
        }

        System.out.println("\n===== RACE FINISHED =====");
    }

    /**
     * Delay method (seconds)
     */
    public static void delay(final int time) {
        try {
            Thread.sleep(time * 1000);
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}
